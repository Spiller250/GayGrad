    }
}

country_event = {
    id = nov_summit_invite.PAS
    title = "Приглашение на саммит"
    desc = "[From.GetNameDef] пригласило Пасифию на саммит бывших государств Великограда."
    picture = GFX_report_event_meeting
    is_target_country = yes  # Ивент показывается только Пасифии (PAS)

    option = {
        name = "Мы приедем"
        add_political_power = -10
        set_country_flag = pas_agrees_to_summit

        name = "Мы не сможем приехать"

    id = nov_summit_all_agree

    desc = "Все приглашенные страны согласились приехать на саммит. Пора начинать обсуждения!"
    picture = GFX_report_event_meeting
    trigger = {
        all_of = {
            has_country_flag = pas_agrees_to_summit
            has_country_flag = flo_agrees_to_summit
            has_country_flag = kiv_agrees_to_summit
            has_country_flag = tok_agrees_to_summit
            has_country_flag = ban_agrees_to_summit
            has_country_flag = tat_agrees_to_summit
            has_country_flag = bnz_agrees_to_summit
        }
    }

    option = {
        name = "Начнем саммит!"
        # Дополнительные эффекты или события могут быть добавлены здесь
    }
}
